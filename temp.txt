let rg_name = "rg-uks-lbg-prod-vr-001"; 
let resource_group = strcat("resourcegroups/", rg_name);
Perf
| where _ResourceId contains resource_group
| where ObjectName == "LogicalDisk"
    and CounterName == "% Free Space"
    and InstanceName has ":"
| project _ResourceId, InstanceName, CounterName, CounterValue
